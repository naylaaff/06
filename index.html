# ========================= # DATASET (ARRAY / SCALABLE) # ========================= data = [ {"id": 1, "nama": "satu"}, {"id": 2, "nama": "dua"}, {"id": 3, "nama": "tiga"}, {"id": 4, "nama": "empat"}, {"id": 5, "nama": "lima"}, {"id": 6, "nama": "enam"}, {"id": 7, "nama": "tujuh"}, {"id": 8, "nama": "delapan"}, {"id": 9, "nama": "sembilan"} ] # ========================= # CORE ENGINE (DRY + REUSE) # ========================= def render(data, renderer): for item in data: renderer(item) # ========================= # VIEW: TABLE # ========================= def view_table(data): print("\n=== TABLE ===") print("ID | NAMA") print("-----------") def row(item): print(f"{item['id']} | {item['nama']}") render(data, row) # ========================= # VIEW: LIST # ========================= def view_list(data): print("\n=== LIST ===") def item_view(item): print(f"- {item['id']} : {item['nama']}") render(data, item_view) # ========================= # VIEW: CARD # ========================= def view_card(data): print("\n=== CARD ===") def card(item): print("-------------") print(f"ID : {item['id']}") print(f"NAMA : {item['nama']}") print("-------------") render(data, card) # ========================= # CONTROLLER (MODULAR) # ========================= def tampilkan_semua(data): view_table(data) view_list(data) view_card(data) # ========================= # MAIN PROGRAM # ========================= def main(): tampilkan_semua(data) if __name__ == "__main__": main()
